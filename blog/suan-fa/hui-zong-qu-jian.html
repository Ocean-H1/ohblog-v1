/**
 * @param {number[]} nums
 * @return {string[]}
 */
var summaryRanges = function(nums) {
    if(nums.length === 1){
        return [nums[0] + '']
    }
    let start=0
    let arr = []
    for(let i = 0;i<nums.length;i++){
        if(nums[i+1] - nums[i] !== 1){
            arr.push(nums.slice(start,i+1))
            start = i+1
        }
    }
    let newArr = []
    for(const item of arr ){
        if(item.length === 1){
            newArr.push(item[0] + '')
        }else {
            newArr.push(item[0]+'->'+item[item.length-1])
        }
    }
    return newArr
};
